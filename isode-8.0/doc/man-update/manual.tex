\def\l@part#1#2{\l@@part{#1}{#2} \let\l@chapter=\l@@@chapter}
\def\l@@@chapter#1#2{%
    \vskip 1.0em plus 1pt
    \nobreak
    \begingroup
	\@tempdima 1.5em
	\parindent \z@ \rightskip \@pnumwidth 
	\parfillskip -\@pnumwidth   
	\bf
	\leavevmode
	#1\hfil \hbox to\@pnumwidth{\hss #2}\par
    \endgroup
    \let\l@chapter=\l@@chapter
}
